def self.evented?

  actor && actor.mailbox.is_a?(Celluloid::IO::Mailbox)

def wait_readable(io)
  io = io.to_io
  if IO.evented?
    mailbox = Thread.current[:celluloid_mailbox]
    mailbox.reactor.wait_readable(io)
  else
    Kernel.select([io])
  end
  nil
end
