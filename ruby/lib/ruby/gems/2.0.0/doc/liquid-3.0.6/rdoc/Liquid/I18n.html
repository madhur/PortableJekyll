def deep_fetch_translation(name)
  name.split('.'.freeze).reduce(locale) do |level, cur|
    level[cur] or raise TranslationError, "Translation for #{name} does not exist in locale #{path}"
  end
end

def interpolate(name, vars)
  name.gsub(/%\{(\w+)\}/) {

    "#{vars[$1.to_sym]}"
  }
