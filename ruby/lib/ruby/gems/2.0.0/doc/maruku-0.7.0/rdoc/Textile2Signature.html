def initialize(s)
        if m = Reg.match(s)
                self.block_name = m[1]
                self.style = m[3]
                self.lang = m[4]
                self.css_class = m[6]
                self.css_id = m[7]
                self.text_align = {nil=>nil,'>'=>'right','<'=>'left',
                        '<>'=>'center','='=>'justified'}[m[8]]
                self.num_left_pad = m[9] && m[9].size
                self.num_right_pad = m[10] && m[10].size
                self.filters = m[11] && m[11].split('|')
                self.double_dot = m[12] && true
        end
end
